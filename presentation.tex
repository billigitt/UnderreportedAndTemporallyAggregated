\protect\hypertarget{why}{}

\protect\hypertarget{why-dont-we-know-this}{}
\begin{itemize}

  Transmission trees are hidden
\item
  We can't see into the future!
\end{itemize}
\end{block}

\begin{frame}{Epi-Estim \footnote<.->{Cori A, Ferguson NM, Fraser C,
  Cauchemez S. A new framework and software to estimate time-varying
  reproduction numbers during epidemics. Am J Epidemiol. 2013}}
\protect\hypertarget{epi-estim-2}{}
If \(R_t \sim \Gamma(a, b)\) and
\(I_t|I_1^{t-1}, R_t \sim \mathrm{Poi}(R_t\sum_{s=1}I_{t-s}w_s)\), then
\[R_t| I_1^t, \boldsymbol{w}\sim \Gamma\left(a+I_t-1, \frac{1}{\sum_{s=1}I_{t-s}w_s+1/b}\right)\]
\end{frame}
